package fr.avenirsesr.portfolio.api.domain.model;

import java.util.List;
import java.util.UUID;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
public class TrackGroup {
  private final UUID id;
  private List<Track> tracks;

  private TrackGroup(UUID id) {
    this.id = id;
    this.tracks = List.of();
  }

  public static TrackGroup create() {
    return new TrackGroup(UUID.randomUUID());
  }

  public static TrackGroup toDomain(UUID id, List<Track> tracks) {
    TrackGroup group = new TrackGroup(id);
    group.setTracks(tracks);
    return group;
  }
}
