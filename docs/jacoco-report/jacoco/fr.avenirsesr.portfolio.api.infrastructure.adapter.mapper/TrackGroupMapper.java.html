package fr.avenirsesr.portfolio.api.infrastructure.adapter.mapper;

import fr.avenirsesr.portfolio.api.domain.model.TrackGroup;
import fr.avenirsesr.portfolio.api.infrastructure.adapter.model.TrackGroupEntity;
import java.util.List;

public interface TrackGroupMapper {
  static TrackGroupEntity fromDomain(TrackGroup group) {
    return new TrackGroupEntity(
        group.getId(), group.getTracks().stream().map(TrackMapper::fromDomain).toList());
  }

  static TrackGroup toDomain(TrackGroupEntity trackGroupEntity) {
    if (trackGroupEntity == null) {
      return null;
    }
    var group = TrackGroup.toDomain(trackGroupEntity.getId(), List.of());
    group.setTracks(
        trackGroupEntity.getTracks().stream()
            .map(trackEntity -> TrackMapper.toDomain(trackEntity, group))
            .toList());
    return group;
  }
}
